    Zubbl SDK Documentation¶

    Version: 0.2.0 | Python: 3.9+ | License: MIT

    Make any AI agent smarter automatically. Drop-in SDK that adds trajectory learning + failure recovery to any agent framework.

    Installation¶

    pip install zubbl-sdk

    Dependencies (installed automatically): httpx, pydantic, python-dotenv

    Quick Start¶

    3-Line Integration¶

    from zubbl import ZubblClient

zubbl = ZubblClient(api_key="zubbl_xxx_xxx")
smart_agent = zubbl.wrap(your_agent)

# Use exactly like before — now it learns and self-recovers
result = smart_agent("Complete this task")

    That's it. Your agent now: - Records every execution as a trajectory - Learns reusable patterns from successes - Auto-recovers from failures (retry, prompt patch, model fallback) - Gets policy recommendations before similar tasks

    ZubblClient¶

    The main entry point. All interactions go through this class.

    Constructor¶

    from zubbl import ZubblClient, RLMode

client = ZubblClient(
    api_key="zubbl_yourtenantid_yoursecret",   # Required
    base_url="https://api.zubbl.in",           # Default
    rl_mode=RLMode.LOCAL,                       # Free tier
    auto_ingest=True,                           # Auto-send trajectories
    auto_query=True,                            # Auto-fetch policies
    auto_recover=True,                          # Auto-recover failures
    timeout=30.0,                               # API timeout (seconds)
    retry_attempts=3,                           # API retry count
    local_cache=True,                           # Cache policies locally
    max_recovery_retries=3,                     # Max failure retries
    fallback_models=["gpt-4o-mini"],            # Fallback LLMs
)
    Parameter Type Default Description
    api_key str required Your Zubbl API key (zubbl_<tenant>_<secret>)
    base_url str "https://api.zubbl.in" API endpoint URL
    rl_mode RLMode LOCAL Learning compute mode
    auto_ingest bool True Automatically send trajectories to Zubbl
    auto_query bool True Automatically fetch policies before tasks
    auto_recover bool True Automatically attempt failure recovery
    timeout float 30.0 API request timeout in seconds
    retry_attempts int 3 Number of API retry attempts
    local_cache bool True Cache policies locally for faster lookups
    max_recovery_retries int 3 Max retries during failure recovery
    fallback_models List[str] [] Fallback LLM models for recovery

    Context Manager¶

    # Sync
with ZubblClient(api_key="zubbl_xxx_xxx") as zubbl:
    smart_agent = zubbl.wrap(my_agent)
    result = smart_agent.run("task")

# Async
async with ZubblClient(api_key="zubbl_xxx_xxx") as zubbl:
    policy = await zubbl.query_async("task")

    Wrapping Agents¶

    wrap()¶

    Wrap any agent to add automatic learning and failure recovery.

    smart_agent = zubbl.wrap(agent, auto_recover=True)
    Parameter Type Default Description
    agent Any required Any AI agent object
    auto_recover bool True Enable auto-recovery for this agent

    Returns: ZubblWrappedAgent — Use exactly like the original agent.

    The wrapped agent transparently proxies all method calls and attributes to the original agent. You can access the original via smart_agent.unwrapped.

    Framework Examples¶

    LangGraph:

    from langchain_openai import ChatOpenAI
from langgraph.prebuilt import create_react_agent
from zubbl import ZubblClient

model = ChatOpenAI(model="gpt-4o")
agent = create_react_agent(model, [your_tools])

zubbl = ZubblClient(api_key="zubbl_xxx_xxx")
smart_agent = zubbl.wrap(agent)

result = smart_agent.invoke({"messages": [("user", "Hello!")]})

    CrewAI:

    from crewai import Crew, Agent, Task
from zubbl import ZubblClient

crew = Crew(agents=[researcher, writer], tasks=[research_task, write_task])

zubbl = ZubblClient(api_key="zubbl_xxx_xxx")
smart_crew = zubbl.wrap(crew)

result = smart_crew.kickoff()

    Custom Agent:

    class MyAgent:
    def run(self, task: str) -> str:
        # your agent logic
        return result

zubbl = ZubblClient(api_key="zubbl_xxx_xxx")
smart_agent = zubbl.wrap(MyAgent())

result = smart_agent.run("Analyze this data")

    Manual Trajectory Recording¶

    For fine-grained control over what Zubbl captures, record trajectories manually instead of using wrap().

    start_trajectory()¶

    Start recording a new trajectory.

    traj_id = zubbl.start_trajectory(
    task_description="Analyze Q4 sales data",
    metadata={"source": "dashboard", "priority": "high"}
)
    Parameter Type Description
    task_description str What the agent is trying to do
    metadata Dict Optional key-value metadata

    Returns: str — Trajectory ID

    record_step()¶

    Record each step the agent takes.

    zubbl.record_step(
    action="search_database",
    tool_name="sql_executor",
    tool_input={"query": "SELECT * FROM sales WHERE quarter = 'Q4'"},
    tool_output="500 rows returned",
    reasoning="Need sales data for Q4 analysis",
    tokens_used=150,
    latency_ms=200
)
    Parameter Type Default Description
    action str required What the agent did
    tool_name str None Tool/function used
    tool_input Dict None Input passed to the tool
    tool_output Any None Output from the tool
    reasoning str None Why the agent chose this action
    tokens_used int 0 Tokens consumed
    latency_ms int 0 Time taken in milliseconds

    end_trajectory()¶

    Finish recording and optionally evaluate the result.

    from zubbl.types import TaskStatus

trajectory = zubbl.end_trajectory(
    status=TaskStatus.SUCCESS,
    actual_output="Q4 revenue: $2.3M, up 15% YoY",
    expected_output="Q4 revenue summary",
    agent_confidence=0.9
)

print(f"Trajectory: {trajectory.trajectory_id}")
print(f"Total tokens: {trajectory.total_tokens}")
    Parameter Type Default Description
    status TaskStatus required Final task status
    actual_output Any None What the agent produced
    expected_output Any None What was expected
    agent_confidence float None Agent's self-assessed confidence (0.0-1.0)

    Returns: Trajectory — The completed trajectory object

    Full Manual Example¶

    from zubbl import ZubblClient
from zubbl.types import TaskStatus

zubbl = ZubblClient(api_key="zubbl_xxx_xxx")

# Start
traj_id = zubbl.start_trajectory(
    task_description="Summarize customer feedback",
    metadata={"agent": "support_bot", "version": "2.1"}
)

# Record steps
zubbl.record_step(
    action="fetch_feedback",
    tool_name="database_query",
    tool_input={"query": "SELECT * FROM feedback WHERE date > '2026-01-01'"},
    tool_output="342 rows",
    tokens_used=100,
    latency_ms=150
)

zubbl.record_step(
    action="analyze_sentiment",
    tool_name="sentiment_analyzer",
    tool_input={"text": "...batch of feedback..."},
    tool_output="positive: 65%, negative: 20%, neutral: 15%",
    tokens_used=800,
    latency_ms=2000
)

zubbl.record_step(
    action="generate_summary",
    tool_name="gpt-4o",
    tool_input={"prompt": "Summarize these sentiment results..."},
    tool_output="Overall customer sentiment is positive...",
    tokens_used=500,
    latency_ms=1500
)

# End
trajectory = zubbl.end_trajectory(
    status=TaskStatus.SUCCESS,
    actual_output="Overall customer sentiment is positive with 65% satisfaction..."
)

    Querying Policies¶

    Before starting a task, ask Zubbl what has worked before.

    query()¶

    policy = zubbl.query(
    task_description="Write unit tests for the login function",
    context={"language": "python", "framework": "pytest"}
)

if policy:
    print(f"Confidence: {policy.confidence_score:.0%}")
    print(f"Recommended steps: {policy.recommended_actions}")
    print(f"Based on {len(policy.source_patterns)} patterns")
else:
    print("No policy yet — Zubbl will learn as you use it")
    Parameter Type Description
    task_description str What the agent is about to do
    context Dict Optional additional context

    Returns: Policy or None

    Async Version¶

    policy = await zubbl.query_async(
    task_description="Debug the auth API",
    context={"error": "timeout on /login"}
)

    Submitting Feedback¶

    Help Zubbl learn faster with explicit feedback on trajectories.

    feedback()¶

    result = zubbl.feedback(
    trajectory_id="traj_abc123",
    rating=0.8,
    comment="Good solution but could be more efficient"
)
    Parameter Type Description
    trajectory_id str ID of the trajectory to rate
    rating float Score from -1.0 (terrible) to 1.0 (perfect)
    comment str Optional feedback text

    Returns: Dict with feedback_id and status

    Validation¶

    Validate a trajectory's output after the fact.

    validate()¶

    result = zubbl.validate(
    trajectory_id="traj_abc123",
    expected_output="Paris",
    actual_output="Paris, France",
    agent_confidence=0.95
)
    Parameter Type Description
    trajectory_id str Trajectory to validate
    expected_output Any Ground truth / expected result
    actual_output Any What the agent actually produced
    agent_confidence float Agent's confidence (0.0-1.0)

    Returns: Dict with validation result

    Failure Recovery¶

    Zubbl automatically intercepts and recovers from agent failures when auto_recover=True.

    How It Works¶

    1. Your agent throws an error or returns a failure
    2. Zubbl classifies the failure (timeout, rate limit, empty response, etc.)
    3. Zubbl applies the best known fix (retry, prompt patch, model fallback, etc.)
    4. If recovery succeeds, the result is returned transparently
    5. Recovery data is recorded so Zubbl learns which fixes work

    Failure Categories¶

    Category Description Example
    EXCEPTION Unhandled error Python exception
    TIMEOUT Request took too long API timeout
    RATE_LIMIT Too many requests 429 response
    CONTEXT_OVERFLOW Input too large Token limit exceeded
    HALLUCINATION Incorrect output Wrong answer
    EMPTY_RESPONSE No output Blank response
    MALFORMED_OUTPUT Bad format Invalid JSON
    API_ERROR Provider error 500 response
    PARTIAL_FAILURE Incomplete result Missing fields

    Fix Types¶

    Fix When Used
    RETRY Transient errors
    RETRY_WITH_BACKOFF Rate limits
    PROMPT_PATCH Learned prompt improvements
    FALLBACK_MODEL Model-specific failures
    CACHED_RESPONSE Known good response exists
    INPUT_TRANSFORM Input format issues
    DECOMPOSE Task too complex
    SKIP Unrecoverable

    Checking Recovery Status¶

    # After a wrapped agent call
recovery = zubbl.last_recovery

if recovery and recovery.success:
    print(f"Recovered using: {recovery.fix_applied}")
    print(f"Attempts: {recovery.attempts}")
    print(f"Original error: {recovery.original_error}")

    Recovery Statistics¶

    stats = zubbl.get_failure_stats()
print(f"Total failures: {stats['total_failures']}")
print(f"Recovered: {stats['total_recovered']}")
print(f"Recovery rate: {stats['recovery_rate']:.0%}")

    Guardrails¶

    Set safety boundaries on agent behavior.

    Adding Guardrails¶

    from zubbl import ZubblClient
from zubbl.types import GuardrailType, GuardrailAction, Guardrail

zubbl = ZubblClient(api_key="zubbl_xxx_xxx")

# Limit max steps per task
zubbl.interceptor  # Access guardrail engine through the client

    Available guardrail types:

    Type Description
    MAX_STEPS Limit number of steps per task
    MAX_TOOL_CALLS Limit tool invocations
    MAX_TOKENS Cap token usage
    MAX_LATENCY_MS Timeout threshold
    REQUIRED_STEP Enforce a required action
    FORBIDDEN_TOOL Block specific tools
    OUTPUT_FORMAT Validate output format
    CUSTOM Custom validation logic

    Actions on violation: WARN, BLOCK, FALLBACK, RETRY

    LLM Adapter¶

    Zubbl includes a model-agnostic LLM adapter. Use it directly if you need multi-provider LLM access.

    Supported Providers¶

    Provider Models
    OpenAI gpt-4o, gpt-4o-mini, gpt-4-turbo, etc.
    Anthropic claude-4-sonnet, claude-4-opus, etc.
    Google gemini-pro, gemini-ultra
    Groq llama-3, mixtral (via OpenAI-compatible API)
    Ollama Any local model
    Together Various open models
    OpenRouter Multi-provider routing

    Usage¶

    from zubbl.llm import create_llm

# OpenAI (default)
llm = create_llm(provider="openai", model="gpt-4o-mini", api_key="sk-...")
response = llm.chat("What is 2 + 2?")
print(response)  # "4"

# Anthropic
llm = create_llm(provider="anthropic", model="claude-4-sonnet", api_key="sk-ant-...")
response = llm.chat("Explain recursion", system="You are a CS tutor")

# Ollama (local)
llm = create_llm(provider="ollama", model="llama3")
response = llm.chat("Hello!")

# Full control with messages
from zubbl.llm import Message

response = llm.generate([
    Message(role="system", content="You are a helpful assistant"),
    Message(role="user", content="Write a haiku about code")
])

print(response.content)        # The haiku
print(response.usage)          # {"prompt_tokens": 20, "completion_tokens": 15}
print(response.latency_ms)     # 450

    Async¶

    response = await llm.generate_async([
    Message(role="user", content="Hello!")

    Error Handling¶

    from zubbl import ZubblClient
from zubbl.types import TaskStatus

zubbl = ZubblClient(api_key="zubbl_xxx_xxx")

try:
    traj_id = zubbl.start_trajectory("risky task")
    # ... do work ...
    zubbl.end_trajectory(TaskStatus.SUCCESS)
except Exception as e:
    # Zubbl still records the failure
    zubbl.end_trajectory(TaskStatus.FAILURE)
    raise

    Complete Example¶

    from zubbl import ZubblClient, RLMode
from zubbl.types import TaskStatus

# Initialize
zubbl = ZubblClient(
    api_key="zubbl_xxx_xxx",
    rl_mode=RLMode.LOCAL,
    auto_recover=True,
    fallback_models=["gpt-4o-mini"]
)

# Option A: Wrap an existing agent (automatic)
smart_agent = zubbl.wrap(my_agent)
result = smart_agent.run("Write a Python sorting function")

# Option B: Manual recording (fine-grained control)
traj_id = zubbl.start_trajectory("Write a Python sorting function")

# Check if Zubbl has a recommendation
policy = zubbl.query("Write a Python sorting function")
if policy:
    print(f"Tip: {policy.recommended_actions}")

# Record steps
zubbl.record_step(
    action="generate_code",
    tool_name="gpt-4o",
    tool_input={"prompt": "Write a merge sort in Python"},
    tool_output="def merge_sort(arr): ...",
    tokens_used=200,
    latency_ms=800
)

zubbl.record_step(
    action="run_tests",
    tool_name="pytest",
    tool_input={"test_file": "test_sort.py"},
    tool_output="3 passed",
    tokens_used=0,
    latency_ms=500
)

# End trajectory
trajectory = zubbl.end_trajectory(
    TaskStatus.SUCCESS,
    actual_output="def merge_sort(arr): ...",
    agent_confidence=0.95
)

# Give feedback
zubbl.feedback(trajectory.trajectory_id, rating=1.0, comment="Clean implementation")

# Check usage
usage = zubbl.get_usage()
print(f"Success rate: {usage.task_success_rate:.0%}")
print(f"Credits remaining: {usage.credits_remaining}")
